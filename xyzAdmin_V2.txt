function GetPlayer(String)
local Found = {}
local strl = tostring(String:lower())
local GetOnce = false
if strl == "all" then
for i,v in pairs(playersService:GetPlayers()) do
table.insert(Found,v.Name)
end
elseif strl == "others" then
for i,v in pairs(playersService:GetPlayers()) do
if v.Name ~= localPlayer.Name then
table.insert(Found,v.Name)
end
end
elseif strl == "me" then
for i,v in pairs(playersService:GetPlayers()) do
if v.Name == localPlayer.Name then
table.insert(Found,v.Name)
end
end
else
if strl~="" then
for i,v in pairs(playersService:GetPlayers()) do
if v.Name:lower()==strl and GetOnce == false then
GetOnce=true
table.insert(Found,v.Name)
end
if v.Name:lower():sub(1, #String) == strl or v.DisplayName:lower():sub(1, #String) == strl and GetOnce == false then
GetOnce=true
table.insert(Found,v.Name)
end
end
end
end
return Found 
end

function xtp(part,position,xp)
local p = part
if p ~= nil and (p:IsA("Part") or p:IsA("BasePart")) then
local ts = game:GetService("TweenService")
local function a(b,c)
    return c*b
end
local d = (a(p.CFrame:ToAxisAngle()) - a(position:ToAxisAngle())).magnitude/math.pi
local tsp = d/tonumber(xp)
local ti = TweenInfo.new(tonumber(tsp), Enum.EasingStyle.Linear)
local tp = {CFrame = position}
ts:Create(p, ti, tp):Play()
end
end

function fsp(F, ...)
assert(type(F) == "function")
local A = {...}
local C = select("#", ...)
local B = Instance.new("BindableEvent")
B.Event:Connect(function()
F(unpack(A, 1, C))
end)
B:Fire()
B:Destroy()
end

function getSoundSource(a,bool)
local isSoundS=true
local sound = a
local sstring
if typeof(a)=="Instance" and sound:IsA("Sound") then
sstring = string.lower(sound.SoundId)
else
sstring = tostring(sound)
isSoundS=false
end
local xstring
local ystring=sstring
if string.find(sstring,"http") then
xstring = string.gsub(sstring,string.lower("http://www.roblox.com/asset/?"),"")
ystring = string.gsub(xstring,"?id=","")
end
if string.find(sstring,"rbx") then
xstring = string.gsub(sstring,string.lower("rbxassetid://"),"")
ystring = xstring
end
local id 
if isSoundS then
id = sound.SoundId:match("%d+")
else
id = sstring:match("%d+")
end
local Asset = game:GetService("MarketplaceService"):GetProductInfo(id)
local source=Asset.Name.." "..ystring
if bool==true then
notify(source)
end
return source
end
